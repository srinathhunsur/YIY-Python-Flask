Authentication and Security 
1. Authentication vs Authorization
• Authentication: confirm who a user is (login).
• Authorization: decide what an authenticated user may do (roles/permissions).

2. Password storage: never store plain text
• Store only a hash of the password (one-way).
• Use a modern, slow, salted hash function (bcrypt, Argon2, PBKDF2).
• Salt prevents identical passwords from producing identical hashes; use per-user salt (most libraries handle this).


3. Hashing algorithms
• Fast hashes (MD5, SHA1, SHA256) → not suitable alone because brute force is fast.
• Use key derivation / slow algorithms: PBKDF2, bcrypt, scrypt, Argon2 (Argon2 is currently best practice).
• Use well-maintained libraries — don’t roll your own.

4. Password verification
• Compare submitted password with stored hash using library provided verify function.
• Use constant-time compare where relevant.

5. Login forms
• Use HTTPS (TLS) to protect credentials in transit.
• Use CSRF protection on forms.
• Validate and sanitize inputs (but password fields should be handled carefully).
• Use rate limiting + account lockout to mitigate brute force.

6. Sessions
• Sessions let the server remember an authenticated user across requests.
• Session may be cookie-based (cookie stores session id or signed data).
• Use secure cookie flags: Secure, HttpOnly, SameSite=strict/lax.
• Set session timeout and automatic logout on inactivity.

7. Session management improvements
• Regenerate session identifier after login (prevents fixation).
• Store minimal sensitive info in session (user id, role).
• Invalidate session on logout.
• Protect against XSS (reduces cookie theft) and CSRF.

8. Extra protections
• Enforce strong password policy and offer password managers.
• Use multi-factor authentication (MFA) for extra security.
• Logging and monitoring for abnormal login patterns.
• Use HTTPS everywhere (HSTS).
• Use Content Security Policy (CSP) and sanitize outputs to reduce XSS risk.

# app.py
-----------------------------------
from flask import Flask, render_template_string, request, redirect, url_for, flash
from flask_sqlalchemy import SQLAlchemy
from flask_login import LoginManager, login_user, logout_user, login_required, current_user, UserMixin
from werkzeug.security import generate_password_hash, check_password_hash
from flask_wtf import FlaskForm
from wtforms import StringField, PasswordField, SubmitField
from wtforms.validators import DataRequired, Length, EqualTo, Regexp

# ----- Config -----
app = Flask(__name__)
app.config['SECRET_KEY'] = 'replace-with-a-secure-random-secret'  # change in runtime
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///site.db'
app.config['SESSION_COOKIE_HTTPONLY'] = True
app.config['SESSION_COOKIE_SECURE'] = True      
# requires HTTPS; set False for local dev if not using HTTPS
app.config['SESSION_COOKIE_SAMESITE'] = 'Lax'   # or 'Strict' depending on needs
app.config['REMEMBER_COOKIE_HTTPONLY'] = True

db = SQLAlchemy(app)
login_manager = LoginManager(app)
login_manager.login_view = 'login'  # redirect unauthorized users here
login_manager.session_protection = "strong"

# ----- Models -----
class User(UserMixin, db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(80), unique=True, nullable=False)
    password_hash = db.Column(db.String(256), nullable=False)  # store hashed password

    def set_password(self, password: str):
        # PBKDF2 with sha256 through werkzeug; you can switch to bcrypt/argon2 via passlib if desired
        self.password_hash = generate_password_hash(password)  

    def check_password(self, password: str) -> bool:
        return check_password_hash(self.password_hash, password)

@login_manager.user_loader
def load_user(user_id):
    return User.query.get(int(user_id))

# ----- Forms -----
class RegisterForm(FlaskForm):
    username = StringField('Username', validators=[DataRequired(), Length(min=3, max=80),
                      Regexp(r'^[A-Za-z0-9_.-]+$', message="Only letters, numbers, _ . - allowed")])
    password = PasswordField('Password', validators=[DataRequired(), Length(min=8)])
    confirm = PasswordField('Confirm', validators=[DataRequired(), EqualTo('password')])
    submit = SubmitField('Register')

class LoginForm(FlaskForm):
    username = StringField('Username', validators=[DataRequired()])
    password = PasswordField('Password', validators=[DataRequired()])
    submit = SubmitField('Sign In')

# ----- Routes -----
@app.route('/')
def index():
    if current_user.is_authenticated:
        return f"Hello, {current_user.username}! <a href='{url_for('logout')}'>Logout</a>"
    return "Hello, Guest! <a href='/login'>Login</a> | <a href='/register'>Register</a>"

@app.route('/register', methods=['GET', 'POST'])
def register():
    form = RegisterForm()
    if form.validate_on_submit():
        if User.query.filter_by(username=form.username.data).first():
            flash("Username already exists", "danger")
            return redirect(url_for('register'))
        user = User(username=form.username.data)
        user.set_password(form.password.data)  # hash stored
        db.session.add(user)
        db.session.commit()
        flash("Registered successfully. Please log in.", "success")
        return redirect(url_for('login'))
    return render_template_string(REG_TEMPLATE, form=form)

@app.route('/login', methods=['GET', 'POST'])
def login():
    form = LoginForm()
    if form.validate_on_submit():
        user = User.query.filter_by(username=form.username.data).first()
        if user and user.check_password(form.password.data):
            login_user(user)  # creates a session
            # IMPORTANT: regenerate session id in a full implementation (Flask-Login handles some)
            flash("Logged in successfully.", "success")
            return redirect(url_for('protected'))
        flash("Invalid username or password", "danger")
    return render_template_string(LOGIN_TEMPLATE, form=form)

@app.route('/logout')
@login_required
def logout():
    logout_user()  # invalidates session
    flash("Logged out.", "info")
    return redirect(url_for('index'))

@app.route('/protected')
@login_required
def protected():
    return f"This is protected. Hello {current_user.username}."

# ----- Simple HTML templates (inlined for demo) -----

<title>Register</title>
<h2>Register</h2>

  {{ form.username.label }} {{ form.username() }}<br>

  {{ form.confirm.label }} {{ form.confirm() }}<br>
  {{ form.submit() }}
</form>
<a href="{{ url_for('login') }}">Login</a>
"""

LOGIN_TEMPLATE = """

<h2>Login</h2>
<form method="post">
  {{ form.hidden_tag() }}
  {{ form.username.label }} {{ form.username() }}<br>
  {{ form.password.label }} {{ form.password() }}<br>
  {{ form.submit() }}
</form>
<a href="{{ url_for('register') }}">Register</a>
"""

# ----- Run -----
if __name__ == '__main__':
    # Create DB tables if not exist (for demo only)
    with app.app_context():
        db.create_all()
    # Run in debug=False for production (and behind HTTPS)
    app.run(debug=True, ssl_context=None)  # for local dev; don't enable SSL here unless configured

How it works (step-by-step)
1. User registration: RegisterForm --> server receives password --> user.set_password() uses generate_password_hash() to create a salted PBKDF2 hash stored in DB.

2. Login: Form submits username/password --> server fetches user, check_password_hash() compares safely --> on success login_user(user) sets Flask-Login session.

3. Session handling: Flask-Login stores session info (user id) in secure cookie; cookie flags set in app config (HttpOnly, Secure, SameSite) to reduce theft.

4. Protected pages: @login_required decorator forces auth; unauthorized users are redirected to the login view.
5. Logout: Calls logout_user() to clear the session.
